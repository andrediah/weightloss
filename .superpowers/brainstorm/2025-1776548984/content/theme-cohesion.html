How Option B works under the hood

Each accent preset ships with a matching surface tint. Swapping accent also swaps the surface tint — one JS call, instant change.

const
ACCENTS = {
  amber: { accent: '#f59e0b', accentDark: '#d97706', surfaceTint: '#fffbf0', textSecondary: '#92400e', border: '#fde68a' },
  teal:   { accent: '#34d399', accentDark: '#059669', surfaceTint: '#f0fdf8', textSecondary: '#065f46', border: '#6ee7b7' },
  rose:   { accent: '#f472b6', accentDark: '#db2777', surfaceTint: '#fff0f6', textSecondary: '#9d174d', border: '#fbcfe8' },
  blue:   { accent: '#60a5fa', accentDark: '#2563eb', surfaceTint: '#eff6ff', textSecondary: '#1e40af', border: '#bfdbfe' },
  violet: { accent: '#a78bfa', accentDark: '#7c3aed', surfaceTint: '#faf5ff', textSecondary: '#5b21b6', border: '#ddd6fe' },
};

function applyAccent(name) {
  const t = ACCENTS[name];
  document.documentElement.style.setProperty('--color-accent', t.accent);
  document.documentElement.style.setProperty('--color-accent-dark', t.accentDark);
  document.documentElement.style.setProperty('--color-surface-primary', t.surfaceTint);
  document.documentElement.style.setProperty('--color-text-secondary', t.textSecondary);
  document.documentElement.style.setProperty('--color-border-default', t.border);
  localStorage.setItem('wlt-accent', name);
}
